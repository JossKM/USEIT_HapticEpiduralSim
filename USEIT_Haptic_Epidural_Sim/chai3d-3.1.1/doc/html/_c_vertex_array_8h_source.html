771  inline void renderInitialize()
772  {
773 #ifdef C_USE_OPENGL
774  // create buffers first time
775  if (m_positionBuffer == (GLuint)(-1))
776  {
777  glGenBuffers(1, &m_positionBuffer);
778  }
779 
780  if ((m_normalBuffer == (GLuint)(-1)) && (m_useNormalData))
781  {
782  glGenBuffers(1, &m_normalBuffer);
783  }
784 
785  if ((m_texCoordBuffer == (GLuint)(-1)) && (m_useTexCoordData))
786  {
787  glGenBuffers(1, &m_texCoordBuffer);
788  }
789 
790  if ((m_colorBuffer == (GLuint)(-1)) && (m_useColorData))
791  {
792  glGenBuffers(1, &m_colorBuffer);
793  }
794 
795  if ((m_tangentBuffer == (GLuint)(-1)) && (m_useTangentData))
796  {
797  glGenBuffers(1, &m_tangentBuffer);
798  }
799 
800  if ((m_bitangentBuffer == (GLuint)(-1)) && (m_useBitangentData))
801  {
802  glGenBuffers(1, &m_bitangentBuffer);
803  }
804 
805  // resize buffers
806  if (m_flagBufferResize)
807  {
808  if (true)
809  {
810  glBindBuffer(GL_ARRAY_BUFFER, m_positionBuffer);
811  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_localPos[0]), GL_STATIC_DRAW);
812  }
813 
814  if (m_useNormalData)
815  {
816  glBindBuffer(GL_ARRAY_BUFFER, m_normalBuffer);
817  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_normal[0]), GL_STATIC_DRAW);
818  }
819 
820  if (m_useTexCoordData)
821  {
822  glBindBuffer(GL_ARRAY_BUFFER, m_texCoordBuffer);
823  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_texCoord[0]), GL_STATIC_DRAW);
824  }
825 
826  if (m_useColorData)
827  {
828  glBindBuffer(GL_ARRAY_BUFFER, m_colorBuffer);
829  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cColorf), &(m_color[0]), GL_STATIC_DRAW);
830  }
831 
832  if (m_useTangentData)
833  {
834  glBindBuffer(GL_ARRAY_BUFFER, m_tangentBuffer);
835  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_tangent[0]), GL_STATIC_DRAW);
836  }
837 
838  if (m_useBitangentData)
839  {
840  glBindBuffer(GL_ARRAY_BUFFER, m_bitangentBuffer);
841  glBufferData(GL_ARRAY_BUFFER, m_numVertices * sizeof(cVector3d), &(m_bitangent[0]), GL_STATIC_DRAW);
842  }
843 
844  m_flagBufferResize = false;
845  }
846 
847  // update buffers if needed
848  if (m_flagPositionData)
849  {
850  glBindBuffer(GL_ARRAY_BUFFER, m_positionBuffer);
851  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_localPos[0]));
852  m_flagPositionData = false;
853  }
854  if (m_flagNormalData)
855  {
856  glBindBuffer(GL_ARRAY_BUFFER, m_normalBuffer);
857  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_normal[0]));
858  m_flagNormalData = false;
859  }
860  if (m_flagTexCoordData)
861  {
862  glBindBuffer(GL_ARRAY_BUFFER, m_texCoordBuffer);
863  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_texCoord[0]));
864  m_flagTexCoordData = false;
865  }
866  if (m_flagColorData)
867  {
868  glBindBuffer(GL_ARRAY_BUFFER, m_colorBuffer);
869  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cColorf), &(m_color[0]));
870  m_flagColorData = false;
871  }
872  if (m_flagTangentData)
873  {
874  glBindBuffer(GL_ARRAY_BUFFER, m_tangentBuffer);
875  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_tangent[0]));
876  m_flagTangentData = false;
877  }
878  if (m_flagBitangentData)
879  {
880  glBindBuffer(GL_ARRAY_BUFFER, m_bitangentBuffer);
881  glBufferSubData(GL_ARRAY_BUFFER, 0, m_numVertices * sizeof(cVector3d), &(m_bitangent[0]));
882  m_flagBitangentData = false;
883  }
884 
885  // bind buffers and set client state
886  {
887  glBindBuffer(GL_ARRAY_BUFFER, m_positionBuffer);
888  glEnableVertexAttribArray(C_VB_POSITION);
889  glVertexAttribPointer(C_VB_POSITION, 3, GL_DOUBLE, GL_FALSE, 0, 0);
890  glVertexPointer(3, GL_DOUBLE, 0, 0);
891  }
892 
893  if (m_useNormalData)
894  {
895  glBindBuffer(GL_ARRAY_BUFFER, m_normalBuffer);
896  glEnableVertexAttribArray(C_VB_NORMAL);
897  glVertexAttribPointer(C_VB_NORMAL, 3, GL_DOUBLE, GL_FALSE, 0, 0);
898  }
899  else
900  {
901  glDisableVertexAttribArray(C_VB_NORMAL);
902  }
903 
904  if (m_useTexCoordData)
905  {
906  glBindBuffer(GL_ARRAY_BUFFER, m_texCoordBuffer);
907  glEnableVertexAttribArray(C_VB_TEXCOORD);
908  glVertexAttribPointer(C_VB_TEXCOORD, 3, GL_DOUBLE, GL_FALSE, 0, 0);
909  }
910  else
911  {
912  glDisableVertexAttribArray(C_VB_TEXCOORD);
913  }
914 
915  if (m_useColorData)
916  {
917  glBindBuffer(GL_ARRAY_BUFFER, m_colorBuffer);
918  glEnableVertexAttribArray(C_VB_COLOR);
919  glVertexAttribPointer(C_VB_COLOR, 4, GL_FLOAT, GL_FALSE, sizeof(cColorf), 0);
920  }
921  else
922  {
923  glDisableVertexAttribArray(C_VB_COLOR);
924  }
925 
926  if (m_useTangentData)
927  {
928  glBindBuffer(GL_ARRAY_BUFFER, m_tangentBuffer);
929  glEnableVertexAttribArray(C_VB_TANGENT);
930  glVertexAttribPointer(C_VB_TANGENT, 3, GL_DOUBLE, GL_FALSE, 0, 0);
931  }
932  else
933  {
934  glDisableVertexAttribArray(C_VB_TANGENT);
935  }
936 
937  if (m_useBitangentData)
938  {
939  glBindBuffer(GL_ARRAY_BUFFER, m_bitangentBuffer);
940  glEnableVertexAttribArray(C_VB_BITANGENT);
941  glVertexAttribPointer(C_VB_BITANGENT, 3, GL_DOUBLE, GL_FALSE, 0, 0);
942  }
943  else
944  {
945  glDisableVertexAttribArray(C_VB_BITANGENT);
946  }
947 
948  glBindBuffer(GL_ARRAY_BUFFER, 0);
949 #endif
950  }

984  inline void allocateData(const int a_numberOfVertices,
985  const bool a_useNormalData,
986  const bool a_useTexCoordData,
987  const bool a_useColorData,
988  const bool a_useTangentData,
989  const bool a_useBitangentData,
990  const bool a_useUserData)
991  {
992  // increment counter
993  m_numVertices = m_numVertices + a_numberOfVertices;
994 
995  // update position data allocation
996  cVector3d pos(0.0, 0.0, 0.0);
997  m_localPos.resize(m_numVertices, pos);
998  m_globalPos.resize(m_numVertices, pos);
999 
1000  // update normal data allocation
1001  m_useNormalData = a_useNormalData;
1002  if (m_useNormalData)
1003  {
1004  cVector3d normal(1.0, 0.0, 0.0);
1005  if ((a_numberOfVertices > 1) || (a_numberOfVertices == 0))
1006  {
1007  m_normal.resize(m_numVertices, normal);
1008  }
1009  else
1010  {
1011  m_normal.push_back(normal);
1012  }
1013  m_flagNormalData = true;
1014  }
1015  else
1016  {
1017  m_normal.clear();
1018  }
1019 
1020  // texture coordinate data allocation
1021  m_useTexCoordData = a_useTexCoordData;
1022  if (m_useTexCoordData)
1023  {
1024  cVector3d texCoord(0.0, 0.0, 0.0);
1025  if ((a_numberOfVertices > 1) || (a_numberOfVertices == 0))
1026  {
1027  m_texCoord.resize(m_numVertices, texCoord);
1028  }
