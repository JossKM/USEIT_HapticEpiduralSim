671 inline int cNumDigits(int a_value)
672 {
673  int digits = 0;
674 
675  if (a_value == 0) { return (1); }
676 
677  double value = (double)(abs(a_value));
678  while (value >= 1.0)
679  {
680  value /= 10;
681  digits++;
682  }
683  return (digits);
684 }

1005  const cVector3d& a_point2,
1006  const double a_epsilon = C_SMALL)
1007 {
1008  // accelerated path for exact equality
1009  if (a_epsilon == 0.0)
1010  {
1011  if ((a_point1(0) == a_point2(0)) && (a_point1(1) == a_point2(1)) && (a_point1(2) == a_point2(2)))
1012  {
1013  return (true);
1014  }
1015  else
1016  {
1017  return (false);
1018  }
1019  }
1020 
1021  if ((fabs(a_point1(0) - a_point2(0)) < a_epsilon) &&
1022  (fabs(a_point1(1) - a_point2(1)) < a_epsilon) &&
1023  (fabs(a_point1(2) - a_point2(2)) < a_epsilon))
1024  {
1025  return (true);

1384 inline double cAngle(const cVector3d& a_vector1,
1385  const cVector3d& a_vector2)
1386 {
1387  // compute length of vectors
1388  double n1 = a_vector1.length();
1389  double n2 = a_vector2.length();
1390  double val = n1 * n2;
1391 
1392  // check if lengths of vectors are not zero
1393  if (fabs(val) < C_SMALL)
1394  {
1395  return (0.0);
1396  }
1397 
1398  // compute angle
1399  double result = a_vector1.dot(a_vector2)/(val);
1400  if (result > 1.0) { result = 1.0; }
1401  else if (result < -1.0) { result = -1.0; }
1402 
1403  return(acos(result));
1404 }
