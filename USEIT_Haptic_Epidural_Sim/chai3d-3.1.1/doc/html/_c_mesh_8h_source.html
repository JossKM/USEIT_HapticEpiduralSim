472 //------------------------------------------------------------------------------
473 // OPERTAOR - (IMPLEMENTED FOR EDGE SORTING PURPOSES)
474 //------------------------------------------------------------------------------
475 inline bool operator<(const cEdge& a_edge0, const cEdge& a_edge1)
476 {
477  double tol = 0.0;
478 
479  cVector3d edge0Vertex0 = a_edge0.m_parent->m_vertices->getLocalPos(a_edge0.m_vertex0);
480  cVector3d edge0Vertex1 = a_edge0.m_parent->m_vertices->getLocalPos(a_edge0.m_vertex1);
481  cVector3d edge1Vertex0 = a_edge1.m_parent->m_vertices->getLocalPos(a_edge1.m_vertex0);
482  cVector3d edge1Vertex1 = a_edge1.m_parent->m_vertices->getLocalPos(a_edge1.m_vertex1);
483 
484 
485  if (cAbs(edge0Vertex0.x() - edge1Vertex0.x()) <= tol)
486  {
487  if (cAbs(edge0Vertex0.y() - edge1Vertex0.y()) <= tol)
488  {
489  if (cAbs(edge0Vertex0.z() - edge1Vertex0.z()) <= tol)
490  {
491  if (cAbs(edge0Vertex1.x() - edge1Vertex1.x()) <= tol)
492  {
493  if (cAbs(edge0Vertex1.y() - edge1Vertex1.y()) <= tol)
494  {
495  return (edge0Vertex1.z() < edge1Vertex1.z());
496  }
497  else
498  {
499  return (edge0Vertex1.y() < edge1Vertex1.y());
500  }
501  }
502  else
503  {
504  return (edge0Vertex1.x() < edge1Vertex1.x());
505  }
506  }
507  else
508  {
509  return (edge0Vertex0.z() < edge1Vertex0.z());
510  }
511  }
512  else
513  {
514  return (edge0Vertex0.y() < edge1Vertex0.y());
515  }
516  }
517  else
518  {
519  return (edge0Vertex0.x() < edge1Vertex0.x());
520  }
521 }
